00020 // This class will set up parallel processes (a thread pool) that will manage chunks.
00021 // This is a singleton class, that manages all threads. The number of allocated threads
00022 // depend on the number of cores of the PC.
00023 //
00024 // There are two types of tasks.
00025 // 1. Load the binary data into a "struct chunk".
00026 // 2. Compute the graphical objects from the binary data.
00027 //
00028 // The reason that these are not always combined into one single step is that it is not
00029 // always the case that the graphical objects are needed. They are only needed if the
00030 // chunk is visible.
00031 // OpenGL data can not be managed from this process, as OpenGL is not thread safe.
00032 //
00033 // It is possible to extend with more types of tasks, not only chunk computations. The requirements
00034 // for extending this mechanisms for other tasks are:
00035 // * It requires a computation that is resource-intensitive
00036 // * The computation can be done in parallel with no dependency on other data
00037 //
00038

00061 // All functions in the public interface are thread safe. They will usually be called from the main process.
00062 class ChunkProcess {
00063 public:
00064         ChunkProcess();
00065         virtual ~ChunkProcess();
00066         void Init(int numThreads); // Create a pool of threads
00067 
00068         // Compute all graphical objects in a chunk. The result has to be polled from GetRecomputedObjects.
00069         void AddTaskComputeChunk(chunk *);
00070 
00071         // Get all new finished objects
00072         void Poll(void);
00073 
00074         // Load new chunk data into a chunk, but do not recompute it. It will replace the previous one
00075         void AddTaskNewChunk(unique_ptr<ChunkBlocks>);
00076 
00077         // Request the processes to terminate
00078         void RequestTerminate(void);
00079 private:
00080         std::condition_variable fCondLock;      // The conditional lock
00081         std::mutex fMutex;              // The mutex used for locking
00082         std::vector<std::thread> fThreads;                      // Identifies the child threads
00083         // Callback function used to initialize threads.
00084         static void *ThreadStatic(ChunkProcess *p);
00085         // Every child thread executes this function, which will not normally return
00086         void Task(void);
00087 
00088         // ==============================================================================================================
00089         // The following variables must always be guarded by 'fMutex' when they are updated. That is because they will be used by
00090         // the main thread as well as the local Task() threads.
00091         //
00092         bool fTerminate;               // The processes has been requested to terminate
00093         std::deque<chunk*> fComputeObjectsInput; // The list of chunk recomputation jobs
00094         std::deque<shared_ptr<ChunkBlocks>> fNewChunksInput; // List of "new chunk" jobs.
00095         // This is where the computed objects are saved. Use a set, to make sure every element is only ever once in it.
00096         std::set<shared_ptr<ChunkObject>> fComputedObjectsOutput;
00097         std::set<shared_ptr<ChunkBlocks>> fNewChunksOutput;
00098         //
00099         // End of list of mutex protected variables.
00100         // ==============================================================================================================
00101 };
