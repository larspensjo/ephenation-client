00031 class TimeMeasure {
00032 public:

00034         void Start() {
00035                 if (!gDebugOpenGL)
00036                         return;
00037                 if (fFirst) {
00038                         glGenQueries(1, &fQuery);
00039                         fFirst = false;
00040                         fTitles[fQuery] = fTitle;
00041                         fResults[fQuery] = 0.0;
00042                 } else {
00043                         GLuint result;
00044                         glGetQueryObjectuiv(fQuery, GL_QUERY_RESULT, &result);
00045                         fResults[fQuery] = result * 0.000001;
00046                 }
00047                 glBeginQuery(GL_TIME_ELAPSED, fQuery);
00048         }
00049         void Stop(void) {
00050                 if (!gDebugOpenGL)
00051                         return;
00052                 glEndQuery(GL_TIME_ELAPSED);
00053         }
