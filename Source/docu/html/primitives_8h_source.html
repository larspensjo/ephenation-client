00039 #define NORMALSCALING 127

00042 //
00043 // This structure defines vertex data used for the first phase of the deferred shading. A voxel based system
00044 // use a lot of data, so it is important to pack it well. The current implementation uses only 20 bytes,
00045 // which includes data for:
00046 // * Vertex coordinate
00047 // * Normal
00048 // * Texture coordiate
00049 // * Light intensity
00050 // * Ambient light
00051 // Specifically, the vertex coordinate can be packed efficiently as it is not the world absolute coordinate. It is the coordinate
00052 // relative to the chunk that the player is in.
00053 struct VertexDataf {
00054 private:
00055         static float fract(float f) { return f; }
00056         static char ConvertNormal(float f) { int ret = int(floorf(f*NORMALSCALING+0.5f)); ASSERT(ret <= 127 && ret >= -128); return ret; }
00057         static short ConvertVertex(float f) { int ret = short(floorf(f*VERTEXSCALING+0.5f)); /* ASSERT(ret <= 32767 && ret >= -32786);*/ return ret; }
00058         static char ConvertTexture(float t) { int ret = char(floorf(fract(t)*TEXTURESCALING+0.5f)); ASSERT(ret <= 127 && ret >= -128); return ret;}
00059         // The normal vector is coded into the first 3 bytes.
00060         // 4:th byte is used for sun intensity and ambient light. Bit 0 to 3 is sun intensity, 4 to 7 is ambient light
00061         char fNormal[4];
00062         short fVertex[3];
00063         char fTexture[2];
00064 public:
00065         void SetTexture(float x, float y) { fTexture[0] = ConvertTexture(x); fTexture[1] = ConvertTexture(y); }
00066         void SetTexture(const glm::vec2 &texture) { this->SetTexture(texture.x, texture.y); }
00067         glm::vec2 GetTexture(void) const { return glm::vec2(fTexture[0]/TEXTURESCALING, fTexture[1]/TEXTURESCALING); }
00068         static void *GetTextureOffset(void) { VertexDataf *p = 0; return &p->fTexture[0]; }
00069         // Take a value from 0 to 255
00070         void SetIntensity(unsigned char intensity) { fNormal[3] = ((intensity>>4) & 0x0F) | (unsigned(fNormal[3]) & 0xF0); }
00071         static void *GetIntensityOffset(void) { VertexDataf *p = 0; return &p->fNormal[3]; }
00072         unsigned char GetIntensity(void) const { return (unsigned(fNormal[3]) & 0x0F) << 4; }
00073         // Take a value from 0 to 255
00074         void SetAmbient(unsigned char ambient) { fNormal[3] = (unsigned(fNormal[3]) & 0x0F) | (ambient & 0xF0); }
00075         // Setting normals for picking mode. We want to transfer data as it is, with no scaling.
00076         void SetNormal(PickingData data) { fNormal[0] = data.rgb[0]; fNormal[1] = data.rgb[1]; fNormal[2] = data.rgb[2]; }
00077         void SetNormal(const glm::vec3 &n) { fNormal[0] = ConvertNormal(n.x); fNormal[1] = ConvertNormal(n.y); fNormal[2] = ConvertNormal(n.z); }
00078         glm::vec3 GetNormal(void) const { return glm::vec3(float(fNormal[0])/NORMALSCALING, float(fNormal[1])/NORMALSCALING, float(fNormal[2])/NORMALSCALING); }
00079         static void *GetNormalOffset(void) { VertexDataf *p = 0; return &p->fNormal[0]; }
00080 
00081         void SetVertex(float x, float y, float z) {
00082                 fVertex[0] = ConvertVertex(x); fVertex[1] = ConvertVertex(y); fVertex[2] = ConvertVertex(z);
00083         }
00084         void SetVertex(const glm::vec3 &v) {
00085                 this->SetVertex(v.x, v.y, v.z);
00086         }
00087         glm::vec3 GetVertex() {
00088                 return glm::vec3(float(fVertex[0])/VERTEXSCALING, float(fVertex[1])/VERTEXSCALING, float(fVertex[2])/VERTEXSCALING);
00089         }
00090         bool LessVertex(const VertexDataf *other) { // Used to make it possible to sort, based on vertex coordinates.
00091                 if (fVertex[0] < other->fVertex[0])
00092                         return true;
00093                 if (fVertex[0] > other->fVertex[0])
00094                         return false;
00095                 if (fVertex[1] < other->fVertex[1])
00096                         return true;
00097                 if (fVertex[1] > other->fVertex[1])
00098                         return false;
00099                 if (fVertex[2] < other->fVertex[2])
00100                         return true;
00101                 return false;
00102         }
00103         static void *GetVertexOffset(void) { VertexDataf *p = 0; return &p->fVertex[0]; }
00104         VertexDataf(const glm::vec3 &normal, const glm::vec2 &texture, const glm::vec3 vertex, unsigned char intensity, unsigned char ambient) {
00105                 this->SetVertex(vertex);
00106                 this->SetNormal(normal);
00107                 this->SetAmbient(ambient);
00108                 this->SetIntensity(intensity);
00109                 this->SetTexture(texture);
00110         }
00111         VertexDataf() {}
00112 };
