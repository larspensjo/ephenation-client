00057 class thread {
00058         pthread_t fThread;
00059         bool valid;
00060 public:
00061         thread() : valid(false) {}
00062         thread(void* (*func)(ChunkProcess *p), ChunkProcess *p) : valid(true) {
00063                 pthread_attr_t attr;
00064                 pthread_attr_init(&attr);
00065                 pthread_create(&fThread, &attr, (void* (*)(void*))func, (void *)p);
00066                 pthread_attr_destroy(&attr);
00067         }
00068         thread(thread &&other) { // Move constructor
00069                 fThread = other.fThread; valid = other.valid; other.valid = false;
00070         }
00071         thread(const thread&) = delete; // Copy constructor, don't allow.
00072         void join(void) {
00073                 assert(valid);
00074                 pthread_join(fThread, NULL);
00075         }
00076         bool joinable(void) const { return valid; }
00077 };
