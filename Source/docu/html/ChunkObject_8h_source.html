00029 // A chunk is 32x32x32 blocks, which can't be used for drawing as it is. It is converted into two
00030 // types: visible blocks and special objects. It is the responsibility of this class to take the chunk
00031 // data and compute what can be drawn from the chunk.
00032 
00033 class ChunkObject {
00034 public:
00035         // The triangle list. There is one list for each block type, as they can't be drawn in
00036         // a single call (using different textures).
00037         // TODO: Should not be public
00038         std::vector<TriangleSurfacef> fVisibleTriangles[256];
00039 
00040         // A boundary box of the chunk
00041         char fBoundXMin, fBoundXMax, fBoundYMin, fBoundYMax, fBoundZMin, fBoundZMax;
00042 
00043         // Get the number of vertices for a block type.
00044         int VertexSize(int bl) const { return fVisibleTriangles[bl].size()*3; }
00045 
00046         // Create and initialize a ChunkObject from a raw chunk. 'pickingmode' is true if the data shall
00047         // be used for picking. 'pdx', etc, are used in picking mode to identify which chunk it is (relative the player).
00048         static unique_ptr<ChunkObject> Make(const chunk *cp, bool pickingMode, int pdx, int pdy, int pdz);
00049 
00050         // True if there are no graphical objects to show in this chunk
00051         bool Empty(void) const;
00052 
00053         // To improve the look, many block types are not drawn as blocks, but as a more advanced 3D model (e.g. trees and lamps).
00054         void FindSpecialObjects(const chunk *cp);
00055 
00056         // For each special object, save information about it.
00057         struct SpecialObject {
00058                 float ambient;         // Amount of ambient light at this object
00059                 unsigned char x, y, z; // The local coordinate of the tree
00060                 unsigned char type;    // Block type
00061         };
00062 
00063         // TODO: These should be std::vector.
00064         // Manage objects found in the chunk. It is a static list, decoded when the chunk is loaded.
00065         // TODO: These should not be public
00066         unique_ptr<SpecialObject[]> fTreeList; int fNumTree;
00067         unique_ptr<SpecialObject[]> fLampList; int fNumLamps;
00068         unique_ptr<SpecialObject[]> fFogList; int fNumFogs;
00069         unique_ptr<SpecialObject[]> fTreasureList; int fNumTreasures;
00070 
00071         // Cyclic pointer back to the chunk the data belongs to. TODO: can this be avoided?
00072         chunk *fChunk;
00073 private:
00074         // Code an array of struct TriangleSurfacef.
00075         // The arguments are used for selection mode.
00076         void FindTriangles(const chunk *cp, bool pickingMode, int pdx, int pdy, int pdz, bool smoothing, bool mergeNormals, bool addNoise);
00077 
00078         ChunkObject(); // Private; only the Make() can create this object
00079 };
