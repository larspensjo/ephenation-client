00092 // This is a coordinate in the Ephenation system, not in the OpenGL system. The chunk
00093 // coordinates are numbered 0, 1, 2, etc.
00094 // This means that real coordinate is chunk coordinate * CHUNK_SIZE.
00095 struct ChunkCoord {
00096         int x, y, z;
00097 
00098         // This operator is used for sorting. Most significant criteria is 'z', to ensure the
00099         // chunks are sorted with high 'z' first.
00100         // TODO: The current sorting criteria is optimized order for drawing chunks for shadows.
00101         //       But the shadow drawing should use it's own sorting criteria.
00102         bool operator<(const ChunkCoord &other) const {
00103                 if (z > other.z) return true;
00104                 else if (z < other.z) return false;
00105 
00106                 if (y < other.y) return true;
00107                 else if (y > other.y) return false;
00108 
00109                 if (x < other.x) return true;
00110                 return false;
00111         }
00112 };
